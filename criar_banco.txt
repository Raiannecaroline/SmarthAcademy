CREATE DATABASE SM_CLIENTE;
USE SMART_ACADEMY;

CREATE TABLE clientes (
    ID INT AUTO_INCREMENT,
    FANTASIA VARCHAR(40),
    CNPJ INT,
    SENHA VARCHAR(16),
    DTCADASTRO date,
    QTPESSOAS INT,
	PRIMARY KEY (ID,CNPJ)
);
CREATE TABLE equipamentos (
    ID INT AUTO_INCREMENT,
    IDCLIENTE int,
    IDEQUIPAMENTO int,
    EQUIPAMENTO VARCHAR(60),
    ULTATU DATETIME,
    EMUSO TINYINT,
    PRIMARY KEY (ID)
);
